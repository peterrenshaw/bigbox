SELECT 
       entry.id, 
       entry.line, 
       entry.length, 
       entry.datetime
FROM   
       entry
ORDER BY 
       entry.datetime DESC;
SELECT status_type.*
FROM   entry,
       entry_status,
       status_type
WHERE
       entry.id = entry_status.entry_id AND
       status_type.id = entry_status.status_id
ORDER BY entry.id, status_type.id;
